**Step 2: Convert thank-you page to Client Component with guard**

The current `app/thank-you/page.tsx` is a Server Component (no `"use client"`). Convert it:

```tsx
"use client";

import { useEffect, useState } from "react";
import { useRouter } from "next/navigation";
import Link from "next/link";
import { getAgents } from "@/lib/content";
import AgentCard from "@/components/contact/AgentCard";
import ScrollFade from "@/components/animations/ScrollFade";

// Agents loaded at module level (they're static JSON — fine for client)
const agents = getAgents();

export default function ThankYouPage() {
  const router = useRouter();
  const [allowed, setAllowed] = useState(false);

  useEffect(() => {
    const flag = sessionStorage.getItem("form_submitted");
    if (!flag) {
      router.replace("/not-found"); // triggers the 404 UI
    } else {
      sessionStorage.removeItem("form_submitted"); // consume the flag
      setAllowed(true);
    }
  }, [router]);

  if (!allowed) return null; // render nothing while guard check runs

  return (
    // ... keep existing JSX exactly as-is ...
  );
}
```

**Important:** The `metadata` export must be removed when converting to `"use client"` — Next.js does not allow metadata exports in Client Components. The page is already `robots: no-index` so this is fine; remove the `export const metadata` block entirely.

## Task 3: reCAPTCHA v3 Verification in /api/contact

**Files:**
- Modify: `app/api/contact/route.ts`

**Context:**
The form client already collects a reCAPTCHA token (env var `NEXT_PUBLIC_RECAPTCHA_SITE_KEY`), but the server route never verifies it against Google's API. Any bot can post directly to `/api/contact`. Add the verification fetch and reject scores below 0.5.

**Step 1: Add reCAPTCHA verification to the route**

In `app/api/contact/route.ts`, after the honeypot check (line 18) and before the field validation (line 22), add:

```typescript
// reCAPTCHA v3 verification
const recaptchaSecret = process.env.RECAPTCHA_SECRET_KEY;
if (recaptchaSecret) {
  const token = body.recaptchaToken as string | undefined;
  if (!token) {
    return NextResponse.json(
      { error: "Missing reCAPTCHA token" },
      { status: 400, headers: CORS_HEADERS }
    );
  }

  const verifyRes = await fetch(
    `https://www.google.com/recaptcha/api/siteverify`,
    {
      method: "POST",
      headers: { "Content-Type": "application/x-www-form-urlencoded" },
      body: `secret=${recaptchaSecret}&response=${token}`,
    }
  );
  const verifyData = (await verifyRes.json()) as {
    success: boolean;
    score: number;
    action: string;
  };

  if (!verifyData.success || verifyData.score < 0.5) {
    // Silently succeed to not tip off bots, but skip email/sheets
    return NextResponse.json({ ok: true }, { status: 200, headers: CORS_HEADERS });
  }
}
```

**Note:** Silently succeeding (instead of returning 400) prevents bots from knowing they were blocked.

**Step 2: Check the contact form sends the token**

Grep for `recaptchaToken` in `components/contact/InquiryForm.tsx`. If not present, add:

```tsx

  process.env.NEXT_PUBLIC_RECAPTCHA_SITE_KEY!,
  { action: "contact_submit" }
);

// Add to the body sent to /api/contact:

```
